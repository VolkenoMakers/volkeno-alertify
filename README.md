## Install

```bash
npm install --save alertify
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'alertify'
import 'alertify/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
